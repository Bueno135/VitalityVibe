CREATE TABLE Cliente (
    ID_Cliente INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    email VARCHAR(100),
    cpf VARCHAR(21),
    senha VARCHAR(100),
    telefone VARCHAR(30),
    dt_nasc DATE,
    cep VARCHAR(15),
    altura VARCHAR(10),
    peso VARCHAR(10),
    sexo VARCHAR(10),
    freq_atv_fisica VARCHAR(100),
    objetivo VARCHAR(100),
    alergias VARCHAR(100),
    problema_saude VARCHAR(100),
    med_controlado VARCHAR(100),
    UNIQUE (ID_Cliente, cpf, email)
);

CREATE TABLE Nutricionista (
    id_nutricionista INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    cpf VARCHAR(21),
    email VARCHAR(100),
    telefone VARCHAR(30),
    senha VARCHAR(100),
    cep VARCHAR(15),
    sexo VARCHAR(15),
    CRN VARCHAR(20),
    formacao VARCHAR(100),
    fk_Especialidade_id_especialidade INT,
    UNIQUE (id_nutricionista, cpf, email)
);

CREATE TABLE Especialidade (
    id_especialidade INT AUTO_INCREMENT PRIMARY KEY ,
    nome_especialidade VARCHAR(100)
);

CREATE TABLE PlanoAlimentar (
    id_plano INT AUTO_INCREMENT PRIMARY KEY,
    nome_pratos VARCHAR(255),
    nome_dieta VARCHAR(255),
    refeicao VARCHAR(100),
    dia_semana VARCHAR(255),
    horario TIME,
    observacoes VARCHAR(255)
);

CREATE TABLE Contrato_Cliente_Nutricionista_PlanoAlimentar (
    id_contrato INT AUTO_INCREMENT PRIMARY KEY, 
    fk_Cliente_ID_Cliente INT,
    fk_Nutricionista_id_nutricionista INT,
    fk_PlanoAlimentar_id_plano INT,
    dt_inic DATE,
    dt_fim DATE
);

CREATE TABLE Mensagem (
    id_mensagem INT AUTO_INCREMENT PRIMARY KEY,
    opcao_conversa VARCHAR(255),
    outro_opcao VARCHAR(255),
    mensagem VARCHAR(100),
    data_envio DATE,
    lida VARCHAR(30),
    motivo VARCHAR(100),
    fk_Cliente_ID_Cliente INT, 
    fk_Nutricionista_id_nutricionista INT
);
