CREATE TABLE Cliente (
    ID_Cliente INT PRIMARY KEY,
    telefone VARCHAR(30),
    dt_nasc DATE,
    altura VARCHAR(10),
    cpf VARCHAR(21),
    sexo VARCHAR(10),
    senha VARCHAR(100),
    freq_atv_fisica VARCHAR(100),
    email VARCHAR(100),
    nome VARCHAR(100),
    alergias VARCHAR(100),
    problema_saude VARCHAR(100),
    med_controlado VARCHAR(100),
    peso VARCHAR(10),
    UNIQUE (ID_Cliente, cpf, email)
);

CREATE TABLE Nutricionista (
    id_nutricionista INT PRIMARY KEY,
    nome VARCHAR(100),
    cpf VARCHAR(21),
    email VARCHAR(100),
    telefone VARCHAR(30),
    senha VARCHAR(100),
    sexo VARCHAR(15),
    formacao VARCHAR(100),
    fk_Especialidade_id_especialidade INT,
    UNIQUE (id_nutricionista, cpf, email)
);

CREATE TABLE Prato (
    id_prato INT PRIMARY KEY UNIQUE,
    nome VARCHAR(100),
    modo_preparo VARCHAR(300)
);

CREATE TABLE Ingredientes (
    id_ingrediente INT PRIMARY KEY UNIQUE,
    gordura FLOAT,
    proteina FLOAT,
    carboidratos FLOAT,
    unidade_medida FLOAT,
    calorias FLOAT,
    fk_CategoriaAlimentar_id_categoria INT
);

CREATE TABLE Especialidade (
    id_especialidade INT PRIMARY KEY UNIQUE,
    nome_especialidade VARCHAR(100)
);

CREATE TABLE CategoriaAlimentar (
    id_categoria INT PRIMARY KEY UNIQUE,
    nome VARCHAR(100)
);

CREATE TABLE Grafico (
    imc FLOAT,
    massa_magra FLOAT,
    BF FLOAT,
    fk_Cliente_ID_Cliente INT
);

CREATE TABLE PlanoAlimentar (
    refeicao VARCHAR(100) PRIMARY KEY,
    dia_semana DATE,
    horario TIME
);

CREATE TABLE Dieta_Cliente_Nutricionista_PlanoAlimentar (
    fk_Cliente_ID_Cliente INT,
    fk_Nutricionista_id_nutricionista INT,
    fk_PlanoAlimentar_refeicao VARCHAR(100),
    dt_inic DATE,
    dt_fim DATE
);

CREATE TABLE Composicao (
    fk_Prato_id_prato INT,
    fk_Ingredientes_id_ingrediente INT,
    Quantidade FLOAT
);

CREATE TABLE PlanoAlimentarPrato (
    fk_PlanoAlimentar_refeicao VARCHAR(100),
    fk_Prato_id_prato INT
);
