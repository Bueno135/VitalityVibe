CREATE TABLE Cliente (
    ID_Cliente INT PRIMARY KEY,
    telefone VARCHAR(50),
    dt_nasc DATE,
    altura VARCHAR(100),
    cpf VARCHAR(21),
    sexo VARCHAR(20),
    senha VARCHAR(255),
    freq_atv_fisica VARCHAR(255),
    email VARCHAR(255),
    nome VARCHAR(255),
    alergias VARCHAR(255),
    problema_saude VARCHAR(255),
    med_controlado VARCHAR(255),
    peso VARCHAR(20),
    UNIQUE (ID_Cliente, email, cpf)
);

CREATE TABLE Nutricionista (
    id_nutricionista INT PRIMARY KEY,
    nome VARCHAR(255),
    cpf VARCHAR(25),
    email VARCHAR(255),
    senha VARCHAR(255),
    sexo VARCHAR(20),
    formacao VARCHAR(255),
    CRN VARCHAR(20),
    telefone VARCHAR(50),
    fk_Especialidade_id_especialidade INT,
    UNIQUE (id_nutricionista, email)
);

CREATE TABLE Prato (
    id_prato INT PRIMARY KEY UNIQUE,
    nome VARCHAR(255),
    modo_preparo VARCHAR(255)
);

CREATE TABLE Ingredientes (
    id_ingrediente INT PRIMARY KEY UNIQUE,
    gordura FLOAT,
    proteina FLOAT,
    carboidratos FLOAT,
    unidade_medida FLOAT,
    calorias FLOAT,
    fk_CategoriaAlimentar_id_categoria INT,
    nome VARCHAR(100)
);

CREATE TABLE Especialidade (
    id_especialidade INT PRIMARY KEY UNIQUE,
    nome_especialidade VARCHAR(255)
);

CREATE TABLE CategoriaAlimentar (
    id_categoria INT PRIMARY KEY UNIQUE,
    nome VARCHAR(255)
);

CREATE TABLE PlanoAlimentar (
    refeicao VARCHAR(255),
    dia_semana VARCHAR(255),
    horario TIME,
    descricao VARCHAR(255),
    id_plano INT PRIMARY KEY UNIQUE
);

CREATE TABLE Contrato_Cliente_Nutricionista_PlanoAlimentar (
    fk_Cliente_ID_Cliente INT,
    fk_Nutricionista_id_nutricionista INT,
    fk_PlanoAlimentar_id_plano INT,
    dt_inic DATE,
    dt_fim DATE,
    id_contrato INT PRIMARY KEY UNIQUE
);

CREATE TABLE Composicao (
    fk_Prato_id_prato INT,
    fk_Ingredientes_id_ingrediente INT,
    Quantidade FLOAT,
    id_composicao INT PRIMARY KEY UNIQUE
);

CREATE TABLE PlanoAlimentarPrato (
    fk_PlanoAlimentar_id_plano INT,
    fk_Prato_id_prato INT
);

CREATE TABLE Mensagem (
    fk_Cliente_ID_Cliente INT,
    fk_Nutricionista_id_nutricionista INT,
    mensagem VARCHAR(255),
    data_envio DATE,
    lida CHAR,
    id_mensagem INT PRIMARY KEY UNIQUE,
    opcao_conversa VARCHAR(50)
);
