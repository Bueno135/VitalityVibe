CREATE TABLE Cliente (
    ID_Cliente INT AUTO_INCREMENT PRIMARY KEY,
    telefone VARCHAR(30),
    dt_nasc DATE,
    altura VARCHAR(10),
    cpf VARCHAR(21),
    sexo CHAR,
    senha VARCHAR(100),
    freq_atv_fisica VARCHAR(20),
    email VARCHAR(100),
    nome VARCHAR(100),
    alergias VARCHAR(100),
    problema_saude VARCHAR(100),
    med_controlado VARCHAR(100),
    UNIQUE (email, cpf)
);

CREATE TABLE Nutricionista (
    id_nutricionista INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    cpf VARCHAR(21),
    email VARCHAR(100),
    telefone VARCHAR(100),
    senha VARCHAR(100),
    sexo CHAR,
    UNIQUE (cpf, email)
);

CREATE TABLE Prato (
    id_prato INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100),
    modo_preparo VARCHAR(300)
);

CREATE TABLE Ingredientes (
    id_ingrediente INT AUTO_INCREMENT PRIMARY KEY,
    gordura FLOAT,
    proteina FLOAT,
    carboidratos FLOAT,
    unidade_medida FLOAT,
    calorias FLOAT,
    fk_CategoriaAlimentar_id_categoria INT
);

CREATE TABLE Especialidade (
    id_especialidade INT AUTO_INCREMENT PRIMARY KEY,
    nome_especialidade VARCHAR(100)
);

CREATE TABLE CategoriaAlimentar (
    id_categoria INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100)
);

CREATE TABLE Grafico (
    imc FLOAT,
    peso INT,
    massa_magra INT,
    BF FLOAT,
    fk_Cliente_ID_Cliente INT
);

CREATE TABLE PlanoAlimentar (
    refeicao VARCHAR(100) PRIMARY KEY,
    dia_semana VARCHAR(20),
    horario TIME
);

CREATE TABLE Formacao (
    id_formacao INT AUTO_INCREMENT PRIMARY KEY,
    nome_formacao VARCHAR(100)
);

CREATE TABLE Dieta_Cliente_Nutricionista_PlanoAlimentar (
    fk_Cliente_ID_Cliente INT,
    fk_Nutricionista_id_nutricionista INT,
    fk_PlanoAlimentar_refeicao VARCHAR(100),
    dt_inicio DATE,
    dt_fim DATE
);

CREATE TABLE Composicao (
    fk_Prato_id_prato INT,
    fk_Ingredientes_id_ingrediente INT,
    Quantidade FLOAT
);

CREATE TABLE Possui_Nutricionista_Especialidade_Formacao (
    fk_Nutricionista_id_nutricionista INT,
    fk_Especialidade_id_especialidade INT,
    fk_Formacao_id_formacao INT
);

CREATE TABLE Possui (
    fk_PlanoAlimentar_refeicao VARCHAR(100),
    fk_Prato_id_prato INT
);
