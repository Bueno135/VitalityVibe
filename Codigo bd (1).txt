CREATE TABLE Cliente (
    ID_Cliente INT PRIMARY KEY,
    name VARCHAR(100),
    telefone INT,
    dt_nasc DATE,

    altura INT,
    cpf VARCHAR(21),
    senha VARCHAR(100),
    freq_atv_fisica CHAR,
    email VARCHAR(50),
    CEP INT,
    alergias VARCHAR(100),
    problema_saude CHAR,
    sexo CHAR,
    fk_Objetivo_id_obj INT
);

CREATE TABLE Nutricionista (
    id_nutricionista INT PRIMARY KEY,
    nome VARCHAR(100),
    cpf INT,
    email VARCHAR(100),
    telefone INT,
    senha VARCHAR(100),
    sexo CHAR,
    CEP INT,
    dt_nasc DATE,
    fk_Especialidade_id_especialidade INT
);

CREATE TABLE Prato (
    id_prato INT PRIMARY KEY,
    nome VARCHAR(100),
    modo_preparo VARCHAR(100)
);

CREATE TABLE Ingredientes (
    id_ingrediente INT PRIMARY KEY,
    gordura FLOAT,
    proteina FLOAT,
    carboidratos FLOAT,
    unidade_medida INT,
    calorias FLOAT,
    fk_CategoriaAlimentar_id_categoria INT
);

CREATE TABLE Especialidade (
    id_especialidade INT PRIMARY KEY,
    nome VARCHAR(100)
);

CREATE TABLE CategoriaAlimentar (
    id_categoria INT PRIMARY KEY,
    nome VARCHAR(100)
);

CREATE TABLE Grafico (
    IMC INT,
    peso INT,
    massa_magra INT,
    BF INT,
    fk_Cliente_ID_Cliente INT
);

CREATE TABLE PlanoAlimentar (
    refeicao VARCHAR(100) PRIMARY KEY,
    dia_semana CHAR,
    horario TIME
);

CREATE TABLE Formacao (
    id_formacao INT PRIMARY KEY,
    nome VARCHAR(100)
);

CREATE TABLE Objetivo (
    id_obj INT PRIMARY KEY,
    nome VARCHAR(100)
);

CREATE TABLE Composicao (

);

CREATE TABLE Possuir (

CREATE TABLE Possui (
